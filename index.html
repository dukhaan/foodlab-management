Request Logger

Kernel-level middleware that tracks and logs every HTTP request flowing through the project.

// Kernel Middleware
class RequestLogger {
  handle($request, $next) {
    Log::info($request->method()
      . ' ' . $request->url());
    return $next($request);
  }
}
